## 트리

### 트리 판별로직

- 루트가 한개인가? 들어오는 간선이 하나도 없는 노드가 정확하게 한개인가? 들어오는 간선이 없는 노드가 2개라면 트리가 아니라 두개의 떨어진 그래프
- 루트를 제외한 모든 노드의 진입간선이 하나인가?
- 모든 노드는 루트로부터 연결이 되어있는가? 노드 개수 V 간선개수 E일때 V - E = 1을 성립하는가? 사이클 판별
- 만약에 진입 차수가 모두 1인데 V = E라면 사이클이 생겼다는 뜻이다.
- https://www.acmicpc.net/problem/6416

## 힙정렬 구현해보기

```c++
#include <iostream>
#include <vector>

using namespace std;

int N, K;
int Count = 0;
int first = -1;
int second = -1;

void heapify(vector<int>& vec, int size, int i)
{
    int smallest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < size && vec[left] < vec[smallest])
    {
        smallest = left;
    }

    if (right < size && vec[right] < vec[smallest])
    {
        smallest = right;
    }

    if (smallest != i)
    {
        Count++;
        if (Count == K)
        {
            first = min(vec[i], vec[smallest]);
            second = max(vec[i], vec[smallest]);
        }
        swap(vec[i], vec[smallest]);
        heapify(vec, size, smallest);
    }
}

void heapSort(vector<int>& vec)
{
    for (int i = N/2-1; i >= 0; i--)
    {
        heapify(vec, N, i);
    }

    for (int i = N-1; i > 0; i--)
    {
        Count++;
        if (Count == K)
        {
            first = min(vec[i], vec[0]);
            second = max(vec[i], vec[0]);
        }
        swap(vec[0], vec[i]);
        heapify(vec, i, 0);
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);


    cin >> N >> K;
    vector<int> vec(N);
    for (int i = 0; i < N; i++)
    {
        cin >> vec[i];
    }
    heapSort(vec);

    // 정렬 후
    if (first == -1)
    {
        cout << -1;
    } else
    {
        cout << first << " " << second;
    }

    return 0;
}
```

```c++
template <typename T>
class MyVector
{
private:
    T* data;
    size_t _size;
    size_t _capacity;

public:
    MyVector(size_t capacity = 2) : _size(0), _capacity(capacity)
    {
        data = new T[_capacity];
    }

    MyVector(const MyVector& other) : _size(other._size), _capacity(other._capacity)
    {
        data = new T[_capacity];
        copy(other.data, other.data + _size, data);
    }

    MyVector& operator=(const MyVector& other)
    {
        if (this != &other)
        {
            delete[] data;
            _size = other._size;
            _capacity = other._capacity;
            data = new T[_capacity];
            copy(other.data, other.data+_size, data);
        }
        return *this;
    }

    ~MyVector()
    {
        delete[] data;
    }

    void push_back(const T& value)
    {
        if (_size == _capacity)
        {
            reserve(_capacity*2);
        }
        data[_size++] = value;
    }

    void pop_back()
    {
        if (_size > 0)
        {
            _size--;
        }
    }

    T& top()
    {
        return data[_size-1];
    }

    void reserve(size_t new_cap)
    {
        if (new_cap <= _capacity) return;
        T* newData = new T[new_cap];
        for (int i = 0; i < _size; i++)
        {
            newData[i] = data[i];
        }

        delete[] data;
        data = newData;
        _capacity = new_cap;
    }

    const T& operator[](size_t index) const
    {
        return data[index];
    }

    size_t size() const {return _size;}
    size_t capacity() const {return _capacity;}
};
```

```c++
#include <iostream>

using namespace std;

struct Node
{
    int data;
    Node* next;

    Node(int val) : data(val), next(nullptr) {}
};


class LinkedList
{
private:
    Node* head;

public:
    LinkedList() : head(nullptr) {}

    void append(int val)
    {
        Node* newNode = new Node(val);
        if (head == nullptr)
        {
            head = newNode;
            return;
        }
        Node* temp = head;
        while (temp->next != nullptr)
        {
            temp = temp->next;
        }
        temp->next = newNode;
    }

    void remove(int val)
    {
        if (head == nullptr) return;

        if (head->data == val)
        {
            Node* temp = head;
            head = head->next;
            delete temp;
            return;
        }

        Node* cur = head;
        while (cur->next != nullptr && cur->next->data != val)
        {
            cur = cur->next;
        }
        if (cur->next != nullptr)
        {
            Node* target = cur->next;
            cur->next = cur->next->next;
            delete target;
        }
    }

    void display()
    {
        Node* temp = head;
        while (temp != nullptr)
        {
            cout << temp->data << " -> ";
            temp = temp->next;
        }
        cout << "NULL" << endl;
    }

    ~LinkedList()
    {
        Node* cur = head;
        while (cur != nullptr)
        {
            Node* next = cur->next;
            delete cur;
            cur = next;
        }
    }
};

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    LinkedList list;
    list.append(10);
    list.append(20);
    list.append(30);

    list.display();

    list.remove(20);
    list.display();

    return 0;
}
```

```c++
class CircularQueue
{
private:
    int* queue;
    int start;
    int end;
    int capacity;

public:
    CircularQueue(int size)
    {
        capacity = size + 1;
        queue = new int[capacity];
        start = 0;
        end = 0;
    }

    bool isFull() {return (end + 1) % capacity == start;}
    bool isEmpty() {return start == end;}

    void push(int val)
    {
        if (isFull())
        {
            cout << "Queue is Full!\n";
            return;
        }
        end = (end + 1) % capacity;
        queue[end] = val;
    }

    int front()
    {
        if (isEmpty()) return -1;
        return queue[(start + 1) % capacity];
    }

    int pop()
    {
        if (isEmpty()) return -1;
        start = (start+1) % capacity;
    }

    ~CircularQueue() {delete[] queue;}
};
```

- push할때 rear+1을 capacity로 나머지연산 만약 앞쪽이 pop으로 빠졌으면 다시 0번째 인덱스부터 들어갈 수 있게 원형으로
- isFull확인시에도 편하게 rear+1로, pop시에는 front를 앞으로 1당겨줌

```c++
class Deque
{
private:
    int* deque;
    int head;
    int tail;
    int capacity;
    int count;

public:
    Deque(int size)
    {
        capacity = size;
        deque = new int[capacity];
        head = 0;
        tail = 0;
        count = 0;
    }

    bool isFull() {return count == capacity;}
    bool isEmpty() {return count == 0;}
    int size() {return count;}


    void push_front(int val)
    {
        if (isFull()) return;
        // 음수 방지 위해
        head = (head-1 + capacity) % capacity;
        deque[head] = val;
        count++;
    }

    void push_back(int val)
    {
        if (isFull()) return;
        deque[tail] = val;
        tail = (tail + 1) & capacity;
        count++;
    }

    void pop_front()
    {
        if (isEmpty()) return;
        head = (head+1) % capacity;
        count--;
    }

    void pop_back()
    {
        if (isEmpty()) return;
        tail = (tail - 1 + capacity) % capacity;
        count--;
    }

    int front()
    {
        if (isEmpty()) return -1;
        return deque[head];
    }

    int back()
    {
        if (isEmpty()) return -1;
        // tail은 미리 다음 자리로 가있으므로 tail-1
        return deque[(tail - 1 + capacity) % capacity];
    }
};
```
