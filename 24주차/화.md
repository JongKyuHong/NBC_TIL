## 배열 {15, 3, 9, 8, 5, 2, 10, 7, 6}에서 짝수만 고려하여 3번째로 큰 수를 출력하세요. (10분)

```c++
#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

int main()
{
    vector<int> vec = {15, 3, 9, 8, 5, 2, 10, 7, 6};
    vector<int> even;

    for (int v : vec)
    {
        if (v % 2 == 0) even.push_back(v);
    }

    auto target = even.end() - 3;
    nth_element(even.begin(), target, even.end());
    cout << *target << endl;

    return 0;
}
```

## 배열 {1, 2, 3, 4}에서 길이가 3인 순열을 모두 출력하세요. 단, 순열의 첫 번째 원소는 항상 1이어야 합니다.(10분)

```c++
#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

int main()
{
    vector<int> vec = {2, 3, 4};
    do
    {
        cout << 1 << " ";
        for (int i = 0; i < 2; ++i)
        {
            cout << vec[i] << " ";
        }
        cout << endl;
    } while (next_permutation(vec.begin(), vec.end()));

    return 0;
}
```

- 이 문제의 경우 사실 3개중에 2개를 뽑는거라 중복이 없어서 이렇게 풀어도 되지만 실제로는 다른식으로 풀어야 할 거 같음

```c++
#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);

    vector<int> vec = {2, 3, 4};
    vector<int> selector = {0, 1, 1};
    int N = vec.size();
    do
    {
        vector<int> picked;
        for (int i = 0; i < N; ++i)
        {
            if (selector[i]) picked.push_back(vec[i]);
        }
        do
        {
            cout << 1 << " "
                << picked[0] << " "
                << picked[1] << endl;
        } while (next_permutation(vec.begin(), vec.end()));
    } while (next_permutation(selector.begin(), selector.end()));

    return 0;
}
```

## 배열 {4, 1, 7, 3, 8, 5}를 5 이상의 값만 내림차순으로 출력하세요. (10분)

```c++
#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);

    vector<int> vec = {4, 1, 7, 3, 8, 5};
    vector<int> tmp;

    for (int v : vec)
    {
        if (v >= 5) tmp.push_back(v);
    }

    sort(tmp.begin(), tmp.end(), greater<int>());

    for (int t : tmp)
    {
        cout << t << " ";
    }
    cout << endl;

    return 0;
}
```
