## 병합 정렬

```c++
#include <iostream>
#include <vector>

using namespace std;

void merge(vector<int>& vec, int left, int mid, int right)
{
    vector<int> tmp;

    int i = left;
    int j = mid + 1;

    while (i <= mid && j <= right)
    {
        if (vec[i] <= vec[j])
        {
            tmp.push_back(vec[i++]);
        }
        else {
            tmp.push_back(vec[j++]);
        }
    }

    while (i <= mid) {
        tmp.push_back(vec[i++]);
    }

    while (j <= right) {
        tmp.push_back(vec[j++]);
    }

    for (int i = 0; i < right - left + 1; i++) {
        vec[i+left] = tmp[i];
    }
}

void mergeSort(vector<int>& vec, int left, int right)
{
    if (left < right)
    {
        int mid = left + (right - left) / 2;
        mergeSort(vec, left, mid);
        mergeSort(vec, mid + 1, right);
        merge(vec, left, mid, right);
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    vector<int> vec = { 9, 3, 1, 5, 13, 12 };
    int n = (int)vec.size();

    cout << "원본 배열: ";
    for (int i = 0; i < n; i++)
        cout << vec[i] << " ";
    cout << endl;

    mergeSort(vec, 0, n - 1);

    cout << "정렬된 배열: ";
    for (int i = 0; i < n; i++)
        cout << vec[i] << " ";
    cout << endl;

    return 0;
}
```

## 힙정렬

```c++
#include <iostream>
#include <vector>

using namespace std;

void heapify(vector<int>& vec, int n, int i) {
    int parent = i;
    int left = parent * 2 + 1;
    int right = parent * 2 + 2;

    if (left < n && vec[left] > vec[parent]) {
        parent = left;
    }

    if (right < n && vec[right] > vec[parent]) {
        parent = right;
    }

    if (parent != i) {
        swap(vec[i], vec[parent]);
        heapify(vec, n, parent);
    }
}

void heapSort(vector<int>& vec, int n) {
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(vec, n, i);
    }

    for (int i = n - 1; i >= 0; i--) {
        swap(vec[0], vec[i]);
        heapify(vec, i, 0);
    }

}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    vector<int> vec = { 4, 10, 3, 5, 1 };
    int n = (int)vec.size();

    heapSort(vec, n);

    for (int i = 0; i < n; i++)
        cout << vec[i] << " ";
    cout << endl;

    return 0;
}
```

## 문자열 내맘대로 정렬하기

```c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int idx;

bool compare(const string& a, const string& b)
{
    if (a[idx] != b[idx])
    {
        return a[idx] < b[idx];
    }
    return a < b;
}

vector<string> solution(vector<string> strings, int n) {
    idx = n;
    sort(strings.begin(), strings.end(), compare);
    return strings;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    // vector<string> result = solution({"sun", "bed", "car"}, 1);
    vector<string> result = solution({"abce", "abcd", "cdx"}, 1);

    for (const string& a : result)
    {
        cout << a << endl;
    }

    return 0;
}
```

## 가장 큰수

```c++
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;

bool compare(const int& a, const int& b)
{
    string avalue = to_string(a);
    string bvalue = to_string(b);
    return (avalue + bvalue) > (bvalue + avalue);
}

string solution(vector<int> numbers) {
    sort(numbers.begin(), numbers.end(), compare);
    string tmp = "";
    for (const int& num : numbers)
    {
        tmp +=  to_string(num);
    }
    if (tmp[0] == '0') return "0";
    return tmp;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    string result = solution({6, 10, 2});

    cout << result;

    return 0;
}
```
