### JKUserInput.cpp

```c++
void UJKUserInput::HandleOnTextCommitted(const FText& Text, ETextCommit::Type CommitMethod)
{
	if (CommitMethod != ETextCommit::OnEnter) return;
	APlayerController* PC = GetOwningPlayer();
	if (IsValid(PC) == false) return;
	AJKPlayerController* JKPC = Cast<AJKPlayerController>(PC);
	if (IsValid(JKPC) == false) return;
	AJKGameStateBase* JKGS = GetWorld()->GetGameState<AJKGameStateBase>();
	if (IsValid(JKGS) == false) return;
	// CurrentTurn과 PlayerState의 PlayerNumber가 다르면 입력 막음
	if (JKGS->GetCurrentTurn() != JKPC->GetPlayerState<AJKPlayerState>()->GetPlayerNumber())
	{
		GEngine->AddOnScreenDebugMessage(-1, 5.0f, FColor::Red, FString::Printf(TEXT("본인턴이 아닙니다.")));
		return;
	}
	JKPC->SetMessage(Text.ToString());
	JKPC->ServerRPCTurnChange();
	EditableTextBox->SetText(FText());
}
```

- UserInput은 서버/클라이언트에 복제되어있는게 아니고 클라이언트에만 존재한다.
- 그래서 턴을 넘기는 신호를 서버로 보내는 ServerRPC를 컨트롤러에 배치하였다. (컨트롤러는 서버에도 존재하므로 ServerRPC보내기 가능)
- GameState와 PlayerState를 비교해서 자신의 차례가 아니면 화면에 붉은색으로 내턴이 아니라고 출력해주었다.

### 컨트롤러

```c++
void AJKPlayerController::ServerRPCTurnChange_Implementation()
{
	AGameModeBase* GM = UGameplayStatics::GetGameMode(this);
	if (IsValid(GM) == false) return;
	AJKGameModeBase* JKGM = Cast<AJKGameModeBase>(GM);
	if (IsValid(JKGM) == false) return;
	AJKGameStateBase* JKGS = GM->GetGameState<AJKGameStateBase>();

}
```

### 게임 스테이트

```c++
void AJKGameStateBase::Tick(float DeltaSeconds)
{
	Super::Tick(DeltaSeconds);

	if (HasAuthority() == false) return;
	TimeAccumulator += DeltaSeconds;

	if (TimeAccumulator >= 1.0f)
	{
		TimeAccumulator -= 1.0f;
		RemainingTime--;
		OnRep_RemainingTime();
	}

	if (RemainingTime <= 0)
	{
		RemainingTime = RemainingTimeInitialValue;
		OnRep_RemainingTime();
		// GameMode에서 턴넘기기
		AGameModeBase* GM = UGameplayStatics::GetGameMode(this);
		if (IsValid(GM) == false) return;
		AJKGameModeBase* JKGM = Cast<AJKGameModeBase>(GM);
		if (IsValid(JKGM) == false) return;
		JKGM->TurnChange();
	}
}
```

- 게임스테이트에서 Tick함수로 시간관리를 하고 있다.
- 서버인경우에만 아래 로직을 수행하게끔 만들었는데 만약에 시간을 모두 흐르게 해서 다음턴으로 넘어가야하는 상황에서는 클라이언트가 Tick아래쪽 로직에 접근이 안되므로 UI가 변경되지 않는 문제가 있었다.
- OnRep을 직접 호출해서 문제를 해결하였다.
- TimerHandle을 쓰는게 더 깔끔하게 만들 수 있는거같은데 Tick에서 만들어서 그냥 두었다.
- RPC랑 Repliacted를 어느정도 이해할 수 있었다.

```c++
void AJKGameModeBase::TurnChange()
{
	AJKGameStateBase* JKGS = GetGameState<AJKGameStateBase>();
	if (IsValid(JKGS) == false) return;
	int32 CurrentTurn = JKGS->GetCurrentTurn();
	CurrentTurn = (CurrentTurn+1) % AllPlayerControllers.Num();
	JKGS->SetCurrentTurn(CurrentTurn);
}
```

## 점프 점프

```c++
// https://www.acmicpc.net/problem/11060

#include <iostream>
#include <vector>
#include <climits>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    cin >> N;
    vector<int> A(N);
    for (int i = 0; i < N; i++)
    {
        cin >> A[i];
    }

    // 가장 왼쪽에서 시작
    int current = 0;
    int count = 0;
    // A[current] 범위안에 어디든 갈 수 있다.
    vector<int> dp(N, INT_MAX);
    dp[0] = 0;

    for (int i = 0; i < N; i++)
    {
        if (dp[i] == INT_MAX) continue;
        for (int j = 1; j <= A[i]; j++)
        {
            if (i+j >= N) break;
            dp[i+j] = min(dp[i+j], dp[i] + 1);
        }
    }
    cout << (dp[N-1] == INT_MAX ? -1 : dp[N-1]);

    return 0;
}
```

- 오랜만에 푸는 dp문제
- 처음에는 이중 벡터로 `dp[현재위치][수행횟수]` 이런식으로 푸려고 했는데 풀다보니 이중이 아닌게 더 낫다는 생각이들어서 바꾸었다.
- 1부터 A[i]만큼의 크기까지 순회하면서 최소값을 dp[i+j]에 쌓는식으로 풀었다.

## 칸토어 집합

```c++
// https://www.acmicpc.net/problem/4779

#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

void find(int l, int r, vector<int> &vec)
{
    if (l == r)
    {
        vec[l] = 1;
        return;
    }
    // 전체 길이 이를 3등분
    int length = (r-l+1)/3;
    // 처음 부분
    find (l, l+length-1, vec);
    find (l+length*2, r, vec);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    int N;
    while (cin >> N)
    {
        int value = (int)pow(3, N);
        vector<int> vec(value, 0);
        // 분할
        find(0, value-1, vec);

        string tmp = "";
        for (int i = 0; i < value; i++)
        {
            tmp += (vec[i] == 1 ? "-" : " ");
        }
        cout << tmp << "\n";
    }

    return 0;
}
```

- find함수 내부에서 length = (r-l)/ 3 부분때문에 애먹었다. 만약에 N이 1이면 2-0/3 이라서 0이되버린다 1이되어야 한다.
  - +1 추가해서 풀게되었다.
- 전체적으로는 더 이상 쪼갤수없을때 vec에 1을주고 나중에 출력할때 vec가 1이면 -를 추가해주는 방식으로 풀었다.
