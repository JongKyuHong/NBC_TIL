# 알고리즘

## dp

팀플끝날때까지는 알고리즘에 많이 신경 못쓸듯??

```c++
// https://www.acmicpc.net/problem/19947

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int H, Y;
    cin >> H >> Y;

    int dp[11];
    dp[0] = H;
    dp[1] = (int)(H*1.05);
    dp[2] = (int)(dp[1]*1.05);
    dp[3] = max({(int)(dp[0]*1.2), (int)(dp[2]*1.05)});
    dp[4] = max({(int)(dp[3]*1.05), (int)(dp[1]*1.2)});
    dp[5] = max({(int)(dp[0]*1.35),(int)(dp[4]*1.05), (int)(dp[2]*1.2)});

    for (int i = 6; i <= 10; i++)
    {
        dp[i] = max({(int)(dp[i-1]*1.05), (int)(dp[i-3]*1.2), (int)(dp[i-5]*1.35)});
    }

    cout << dp[Y];

    return 0;
}
```

- https://www.acmicpc.net/problem/19947
- 쉬운 dp문제 이렇게 풀다가 나중에는 for문안에서 i가 3이상일때랑 i가 5이상일때 체크해서 더 깔끔하게 로직변경

```c++
int dp[11];
dp[0] = H;
for (int i = 1; i <= 10; i++)
{
    dp[i] = (int)(dp[i-1]*1.05);

    if (i >= 3)
    {
        dp[i] = max(dp[i], (int)(dp[i-3]*1.2));
    }

    if (i >= 5)
    {
        dp[i] = max(dp[i],(int)(dp[i-5]*1.35));
    }
}
```
